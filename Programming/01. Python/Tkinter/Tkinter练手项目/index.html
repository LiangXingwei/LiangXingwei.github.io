    Tkinter练手项目

    此项目为红旗 E115 选择车型 Excel 配置文件检测工具项目，该工具用于检测 Excel 文件内容，使用 C++ 实现较为麻烦，故考虑使用有较多第三方库的 python 重新实现。

    程序界面使用 Tkinter 实现，Excel 文件操作使用的是 [[xlwings 实践|xlwings]] 。

    目录结构： 

    \src

|____`main.py`

|____`gui.py`

|____`file_checker.py`

|____translator.py

    实现代码如下：

    main.py 

    #!/usr/bin/env python

import gui

def main() -> None:
    gui.Gui()

if __name__ == '__main__':
    main()

    gui.py 

    '''
定义程序主界面
'''

from datetime import datetime
import tkinter as tk
import tkinter.ttk as ttk
import tkinter.messagebox
import tkinter.filedialog
import file_checker
import translator
import threading

class Gui(object):
    '''
    主窗口类
    '''
    def __init__(self) -> None:
        self.__root = tk.Tk()
        self.__root.title('红旗选装车型辅助工具')
        # TODO: 设置窗口图标

        # 设定窗口的大小(长 * 宽)，并居中显示
        width = 720
        height = 460
        geometry_str = '{}x{}+{}+{}'.format(width, 
                                            height, 
                                            (self.__root.winfo_screenwidth() - width) // 2,
                                            (self.__root.winfo_screenheight() - height) // 2)

        self.__root.geometry(geometry_str)  # 这里的乘是小x
        self.__root.wm_minsize(width, height)

        # 创建NOTEBOOK
        self.__note_book = ttk.Notebook(self.__root)
        self.__file_check_frame = ttk.Frame(self.__note_book)
        self.__translate_frame = ttk.Frame(self.__note_book)
        self.__note_book.add(self.__file_check_frame, text='配置文件检测')
        self.__note_book.add(self.__translate_frame, text='配置文件翻译')
        #self.__note_book.select(self.__translate_frame)

        # ------------------------------------------ 文件检测界面 ------------------------------------------------------------
        # 创建 tkinter 字符变量
        self.__filepath = tk.StringVar()
        # 创建文件路径编辑框，并与变量绑定
        self.__path_entry = tk.Entry(self.__file_check_frame, textvariable=self.__filepath)

        # 创建“选择文件”按钮
        self.__btn_browse = tk.Button(self.__file_check_frame, text='选择文件', command=self.__on_click_btn_browse)
        self.__btn_browse.focus()

        # 创建“开始检测”按钮
        self.__btn_start_check = tk.Button(self.__file_check_frame, text='开始检测', command=self.__on_click_btn_start_check)

        # 创建“导出检测记录”按钮
        self.__btn_export = tk.Button(self.__file_check_frame, text='导出检测记录', command=self.__on_btn_export, state='disabled')

        # 创建进度条
        self.__progress = tk.DoubleVar()
        self.__progressbar = ttk.Progressbar(self.__file_check_frame, maximum=100.0, variable=self.__progress)

        # 创建提示 label
        self.__tip_on_label = tk.StringVar()
        self.__label = tk.Label(self.__file_check_frame, textvariable=self.__tip_on_label, anchor='w')

        # 创建文字信息显示区域
        self.__text_area = tk.Text(self.__file_check_frame, yscrollcommand='auto')

        # 创建滚动条
        self.__scroll = tk.Scrollbar(self.__file_check_frame, orient='vertical')

        # 文本框和滚动条关联
        self.__scroll.config(command=self.__text_area.yview)
        self.__text_area.config(yscrollcommand=self.__scroll.set)
        # 记录工作线程ID
        self.__work_thread = None

        # -------------------------------------------------------------------------------------------------------------------


        # ------------------------------------------ 文件翻译界面 ------------------------------------------------------------
        # 创建 label
        self.__config_path_label = tk.Label(self.__translate_frame, text='配置表路径：', anchor='e')

        # 创建 tkinter 字符变量
        self.__config_filepath = tk.StringVar()
        # 创建文件路径编辑框，并与变量绑定
        self.__config_path_entry = tk.Entry(self.__translate_frame, textvariable=self.__config_filepath)

        # 创建“选择文件”按钮
        self.__btn_config_browse = tk.Button(self.__translate_frame, text='选择文件', command=self.__on_click_btn_config_browse)
        self.__btn_config_browse.focus()

        # 创建 label
        self.__trans_path_label = tk.Label(self.__translate_frame, text='中英文对照表路径：', anchor='e')

        # 创建 tkinter 字符变量
        self.__trans_filepath = tk.StringVar()
        # 创建文件路径编辑框，并与变量绑定
        self.__trans_path_entry = tk.Entry(self.__translate_frame, textvariable=self.__trans_filepath)

        # 创建“选择文件”按钮
        self.__btn_trans_browse = tk.Button(self.__translate_frame, text='选择文件', command=self.__on_click_btn_trans_browse)

        # 创建进度条
        self.__progress_trans = tk.DoubleVar()
        self.__progressbar_trans = ttk.Progressbar(self.__translate_frame, maximum=100.0, variable=self.__progress_trans)

        # 创建“开始翻译”按钮
        self.__btn_trans_trans = tk.Button(self.__translate_frame, text='开始翻译', command=self.__on_click_btn_start_translate)

        # 创建文字信息显示区域
        self.__trans_output = tk.Text(self.__translate_frame, yscrollcommand='auto')

        # 创建滚动条
        self.__trans_output_scroll = tk.Scrollbar(self.__translate_frame, orient='vertical')

        # 文本框和滚动条关联
        self.__trans_output_scroll.config(command=self.__trans_output.yview)
        self.__trans_output.config(yscrollcommand=self.__trans_output_scroll.set)

        # 创建“导出检测记录”按钮
        self.__btn_trans_export = tk.Button(self.__translate_frame, text='导出翻译记录', command=self.__on_btn_trans_export, state='disabled')
        # 记录工作线程ID
        self.__trans_thread = None

        # -------------------------------------------------------------------------------------------------------------------

        # 绑定界面变化的处理函数
        self.__root.bind('<Configure>', self.__on_resize)
        self.__root.protocol('WM_DELETE_WINDOW', self.__on_close)

        # 进入事件循环
        self.__root.mainloop()

    def __on_resize(self, event:tk.Event) -> None:
        '''
        处理窗口尺寸改变时界面控件的变化，窗口创建时也会触发此函数，所以只在这里做界面的布局就可以了
        '''
        width = self.__root.winfo_width()
        height = self.__root.winfo_height()
        self.__note_book.place(x=0, y=0, width=width, height=height)
        frame_with = 0
        frame_height = 0

        cur_tab = self.__note_book.select()
        #print(type(cur_tab), self.__file_check_frame.__str__())
        if self.__file_check_frame.__str__() == cur_tab:
            frame_with = self.__file_check_frame.winfo_width()
            frame_height = self.__file_check_frame.winfo_height()
        else:
            frame_with = self.__translate_frame.winfo_width()
            frame_height = self.__translate_frame.winfo_height()

        self.__resize_file_check_widgets(frame_with, frame_height)
        self.__resize_file_trans_widgets(frame_with, frame_height)

    def __resize_file_check_widgets(self, width:int, heigth:int) -> None:
        '''
        窗口尺寸改变时，重新布局文件检测界面的控件
        '''
        self.__path_entry.place(x=10, y=10, width=width - 40 - 88 * 2, height=25, anchor='nw')
        self.__btn_browse.place(x=width - 108, y=10, width=88, height=25, anchor='ne')
        self.__btn_start_check.place(x=width - 10, y=10, width=88, height=25, anchor='ne')
        self.__btn_export.place(x=width - 10, y=heigth - 35, width=88, height=25, anchor='ne')
        self.__progressbar.place(x=10, y=45, width=self.__path_entry.winfo_width(), height=25, anchor='nw')
        self.__label.place(x=width-10, y=45, width=186, height=25, anchor='ne')
        self.__text_area.place(x=10, y=80, width=width - 25, height=heigth - 125, anchor='nw')
        self.__scroll.place(x=width - 10, y=80, width=15, height=heigth - 125, anchor='ne')

    def __resize_file_trans_widgets(self, width:int, heigth:int) -> None:
        '''
        窗口尺寸改变时，重新布局文件翻译界面的控件
        '''
        self.__config_path_label.place(x=10, y=10, width=100, height=25, anchor='nw')
        self.__config_path_entry.place(x=110, y=10, width=width - 218, height=25, anchor='nw')
        self.__btn_config_browse.place(x=width - 10, y=10, width=88, height=25, anchor='ne')
        self.__trans_path_label.place(x=10, y=45, width=100, height=25, anchor='nw')
        self.__trans_path_entry.place(x=110, y=45, width=width - 218, height=25, anchor='nw')
        self.__btn_trans_browse.place(x=width - 10, y=45, width=88, height=25, anchor='ne')
        self.__progressbar_trans.place(x=10, y=80, width=width - 118, height=25, anchor='nw')
        self.__btn_trans_trans.place(x=width - 10, y=80, width=88, height=25, anchor='ne')
        self.__btn_trans_export.place(x=width - 10, y=heigth - 35, width=88, height=25, anchor='ne')
        self.__trans_output.place(x=10, y=115, width=width - 25, height=heigth - 160, anchor='nw')
        self.__trans_output_scroll.place(x=width - 10, y=115, width=15, height=heigth - 160, anchor='ne')

    def __on_close(self):
        '''
        处理窗口关闭事件
        '''
        if self.__is_thread_working():
            tkinter.messagebox.showinfo('配置文件检测', message='正在检测文件，请稍候...')
        elif self.__is_trans_thread_working():
            tkinter.messagebox.showinfo('配置文件翻译', message='正在翻译文件，请稍候...')
        else:
            self.__root.destroy()

    def __on_click_btn_browse(self, *args) -> None:
        '''
        “选择文件”按钮响应函数
        '''
        filepath = tkinter.filedialog.askopenfilename(filetypes=[('Excel', ['.xls', '.xlsx'])])
        if '' != filepath:
            self.__filepath.set(filepath)

    def __on_click_btn_config_browse(self, *args) -> None:
        '''
        文件翻译界面，配置文件选择按钮响应函数
        '''
        filepath = tkinter.filedialog.askopenfilename(filetypes=[('Excel', ['.xls', '.xlsx'])])
        if '' != filepath:
            self.__config_filepath.set(filepath)

    def __on_click_btn_trans_browse(self, *args) -> None:
        '''
        文件翻译界面，中英文对照文件选择按钮响应函数
        '''
        filepath = tkinter.filedialog.askopenfilename(filetypes=[('Excel', ['.xls', '.xlsx'])])
        if '' != filepath:
            self.__trans_filepath.set(filepath)

    def __on_click_btn_start_check(self, *args) -> None:
        '''
        “开始检测”按钮响应函数
        '''
        if '' == self.__filepath.get():
            tkinter.messagebox.showinfo('配置文件检测', message='请先选择一个Excel文件！')
            return

        if self.__is_thread_working():
            tkinter.messagebox.showinfo('配置文件检测', message='正在检测文件，请稍候...')
            return

        self.__work_thread = threading.Thread(target=self.__start_work_thread)
        self.__work_thread.start()

        self.__tip_on_label.set('正在检测文件，请稍候...')

        # “导出检测记录”按钮设为可用
        self.__btn_export.configure(state='normal')

    def __on_click_btn_start_translate(self, *args) -> None:
        '''
        “开始翻译”按钮响应函数
        '''
        if '' == self.__config_filepath.get():
            tkinter.messagebox.showinfo('配置文件翻译', message='请先选择配置表文件！')
            return

        if '' == self.__trans_filepath.get():
            tkinter.messagebox.showinfo('配置文件翻译', message='请先选择中英文对照表文件！')
            return

        if self.__is_trans_thread_working():
            tkinter.messagebox.showinfo('配置文件翻译', message='正在检测文件，请稍候...')
            return

        self.__trans_thread = threading.Thread(target=self.__start_trans_thread)
        self.__trans_thread.start()

        # “导出翻译记录”按钮设为可用
        self.__btn_trans_export.configure(state='normal')

    def __on_btn_export(self, *args) -> None:
        '''
        “导出检测记录”按钮响应函数
        '''
        if self.__is_thread_working():
            tkinter.messagebox.showinfo('配置文件检测', message='正在检测文件，请稍候...')
            return

        filepath = tkinter.filedialog.asksaveasfilename(defaultextension='.txt', filetypes=[('文本文件', ['.txt'])])
        if '' != filepath:
            # print('save as {}'.format(filepath))
            buf = self.__text_area.get('1.0', 'end')
            with open(filepath, 'w') as file:
                file.write(buf)

    def __on_btn_trans_export(self, *args) -> None:
        '''
        “导出翻译记录”按钮响应函数
        '''
        if self.__is_trans_thread_working():
            tkinter.messagebox.showinfo('配置文件翻译', message='正在翻译文件，请稍候...')
            return

        filepath = tkinter.filedialog.asksaveasfilename(defaultextension='.txt', filetypes=[('文本文件', ['.txt'])])
        if '' != filepath:
            buf = self.__trans_output.get('1.0', 'end')
            with open(filepath, 'w') as file:
                file.write(buf)

    def __set_progress(self, progress:float=0.0) -> None:
        '''
        设置进度条
        '''
        self.__progress.set(progress)

    def __set_trans_progress(self, progress:float=0.0) -> None:
        '''
        设置翻译进度条
        '''
        self.__progress_trans.set(progress)

    def __init_text(self) -> None:
        '''
        初始化文字信息显示
        '''
        # 清空文本框
        self.__text_area.replace('1.0', 'end', '')

        msg = '[{}]开始检测...\n'.format(datetime.now())
        msg += '[{}]目标文件：{}\n'.format(datetime.now(), self.__filepath.get())

        self.__text_area.insert('end', msg)

    def __init_trans_text(self) -> None:
        '''
        初始化翻译文字信息显示
        '''
        self.__trans_output.replace('1.0', 'end', '')

        msg = '[{}]开始翻译...\n'.format(datetime.now())
        msg += '[{}]配置表文件：{}\n'.format(datetime.now(), self.__config_filepath.get())
        msg += '[{}]中英文对照表文件：{}\n'.format(datetime.now(), self.__trans_filepath.get())

        self.__trans_output.insert('end', msg)

    def __set_info_text(self, info:str, show_msg_box:bool=False) -> None:
        '''
        设置显示信息
        '''
        if not info.isspace():
            self.__text_area.insert(index='end', chars=info)

            # 文本框滚动到末行
            self.__text_area.see(index='end')

        if show_msg_box:
            tkinter.messagebox.showinfo(title='配置文件检测', message=info)

    def _set_trans_output_text(self, info:str, show_msg_box:bool=False) -> None:
        '''
        设置翻译显示信息
        '''
        if not info.isspace():
            self.__trans_output.insert(index='end', chars=info)
            self.__trans_output.see(index='end')

        if show_msg_box:
            tkinter.messagebox.showinfo(title='配置文件翻译', message=info)

    def __start_work_thread(self) -> None:
        '''
        启动工作线程
        '''
        self.__init_text()
        check_op = file_checker.FileChecker(self.__filepath.get(), self.__set_progress, self.__set_info_text)
        check_op.start_check()
        self.__tip_on_label.set('检测完成')

    def __start_trans_thread(self) -> None:
        '''
        启动翻译线程
        '''
        self.__init_trans_text()
        trans_op = translator.Translator(self.__config_filepath.get(), self.__trans_filepath.get(), self.__set_trans_progress, self._set_trans_output_text)
        trans_op.start_translate()

    def __is_thread_working(self) -> bool:
        '''
        检查工作线程是否正在执行，是则返回True，否则返回False
        '''
        if self.__work_thread and self.__work_thread.is_alive():
            return True

        return False

    def __is_trans_thread_working(self) -> bool:
        '''
        检查翻译线程是否正在执行，是则返回True，否则返回False
        '''
        if self.__trans_thread and self.__trans_thread.is_alive():
            return True

        return False

    '''

import xlwings as xw
import pythoncom
from datetime import datetime

class FileChecker(object):
    '''
    文件检测类，完成 Excel 配置文件的检测功能
    '''

    def __init__(self, filepath:str, progress_callback=None, info_callback=None) -> None:
        '''
        构造函数

        参数说明：progress_callback，外部传入的用于设置进度的函数，应当接收一个 float 型的参数，有效值为 0.0~100.0
                 函数原型：set_progress(progress:float=0.0)

                 info_callback，外部传入的用于显示信息的函数，应当接收一个 str 型的参数和一个 bool 型的参数
                 函数原型：show_info(info:str='', show_message_box:bool=false)
        '''

        #self.__app = xw.App(visible=False, add_book=False)
        #self.__book = self.__app.books.open(filepath, read_only=True)

        #self.__cols_count = self.__used_range.columns.count
        pythoncom.CoInitialize()
        app = xw.App(visible=False, add_book=False)
        book = app.books.open(filepath, read_only=True)
        sheet = book.sheets[0]
        sheet.activate()
        used_range = sheet.used_range

        self.__app_pid = app.pid
        self.__rows_count = used_range.rows.count
        self.__cols_count = used_range.columns.count
        self.__progress_callback = progress_callback
        self.__info_callback = info_callback
        self.__error_count = 0

    def __del__(self) -> None:
        '''
        析构函数，释放相关资源
        '''
        #self.__book.close()
        #self.__app.quit()
        #self.__app.books.active.close()
        #self.__app.kill()
        app = xw.apps[self.__app_pid]
        book = app.books.active
        book.close()
        app.kill()
        pythoncom.CoUninitialize()

    def start_check(self) -> bool:
        '''
        开始进行文件检测
        返回值：检测未发现错误返回true；否则返回false
        '''

        if not self.__check_header():
            return False

        self.__check_vehicle_data()
        self.__summary()

        return True

    def __check_header(self) -> bool:
        '''
        检测Excel表格表头
        返回值：检测未发现错误返回true；否则返回false
        '''
        if not self.__check_header_vehicle_code():
            return False

        if not self.__check_header_ecu_name():
            return False

        if not self.__check_ecu_info_col():
            return False

        return True

    def __check_header_vehicle_code(self) -> bool:
        '''
        检测表头车辆规格码（即第 1 行第 1 列，车辆规格码必须在这个单元格，否则视为错误）
        返回值：检测未发现错误返回true；否则返回false
        '''
        if '车型规格号' != xw.Range(('A1')).value:
            self.__error_count += 1
            self.__info_callback('[{}]表头检测错误！\n车型规格码必须为表格的第 1 行第 1 列！\n'.format(datetime.now()), True)
            return False

        return True

    def __check_header_ecu_name(self) -> bool:
        '''
        检测表头控制器名称，控制器名称所在的单元格是3个单元格合并而成的一个大单元格
        返回值：检测未发现错误返回true；否则返回false
        '''
        first_row = xw.Range((1, 3), (1, self.__cols_count))
        error = False
        cell_pos = ''
        ecu_name = ''

        # ECU数据从第3列开始，当列索引为3的倍数时，取到有效名称，否则应为无效单元格
        for cell in first_row:
            cell_pos = cell.get_address(False, False)
            if cell.merge_cells:
                if 'None' != str(cell.value).strip():
                    ecu_name = str(cell.value).strip()
                else:
                    if 0 ==  cell.column % 3:
                        error = True
                        break
            else:
                error = True
                break

            self.__progress_callback(1 / self.__rows_count * 100)

        if error:
            msg = '[{}]表头检测错误！\n控制器 {} 信息格式错误，请检查 {} 单元格！\n'.format(datetime.now(), ecu_name, cell_pos)
            self.__info_callback(msg, True)
            return False

        return True

    def __check_ecu_info_col(self) -> bool:
        '''
        检测表头控制器信息，即零件号、软件和软件版本号、配置码
        返回值：检测未发现错误返回true；否则返回false
        '''
        second_row = xw.Range((2, 3), (2, self.__cols_count))
        error = False
        cell_pos = ''

        # ECU数据从第3列开始，每3列为一组，依次为零件号、软件号和软件版本号、配置码
        for cell in second_row:
            cell_str = str(cell.value).strip()
            cell_pos = cell.get_address(False, False)
            if ((0 == (cell.column % 3)) and ('零件号' != cell_str)) or \
                ((1 == (cell.column % 3)) and ('软件号和软件版本号' != cell_str)) or \
                    ((2 == (cell.column % 3)) and ('配置码' != cell_str)):
                        error = True
                        break

            self.__progress_callback(2 / self.__rows_count * 100)

        if error:
            msg = '[{}]表头检测错误！\n表格格式错误，请检查 {} 单元格！\n'.format(datetime.now(), cell_pos)
            self.__info_callback(msg, True)
            return False

        return True

    def __check_vehicle_data(self) -> bool:
        '''
        逐行检测车型配置内容
        返回值：检测未发现错误返回true；否则返回false
        '''
        all_range = xw.Range((3, 1), (self.__rows_count, self.__cols_count))

        # 车型数据从第3行开始，每个车型的具体数据信息为一行
        for row in all_range.rows:
            vehicle_code = str(row[0].value).strip()
            if 'None' == vehicle_code:
                continue

            vehicle_name = str(row[1].value).strip()
            if 'None' == vehicle_name:
                msg = '[{}]车辆规格码 {} 对应车型名称为空！请检查 {} 单元格！\n'.format(datetime.now(), vehicle_code, row[1].get_address(False, False))
                self.__error_count += 1
                self.__info_callback(msg)

            for col_index in range(3, self.__cols_count + 1):
                self.__check_cell_data(vehicle_name, row, col_index)

            self.__progress_callback(row.row / self.__rows_count * 100)

        return True

    def __check_cell_data(self, vehicle_name:str, row:xw.Range, col_index:int) -> None:
        '''
        检测目标单元格的数据，若检测到错误，记录错误信息，目前有两种错误情况：
        1.某一个控制器的“软件号和软件版本号”有内容，而“零件号”为空；
        2.单元格中的字符串包含"ERROR"子串。
        '''
        cell_data = str(row[col_index - 1].value).strip()

        # 软件号和软件版本号
        if 1 == (col_index % 3):
            # 检查零件号是否为空
            part_number = str(row[col_index - 2].value).strip()
            if (('' == part_number) or ('None' == part_number)) and \
                ('' != cell_data):
                    ecu_name = str(xw.Range((1, col_index - 1)).value).strip()
                    msg = '[{}]{} {} 零件号为空！请检查 {} 单元格！\n'.format(datetime.now(), 
                                                                 vehicle_name, ecu_name, 
                                                                 xw.Range((row[col_index -2].row, col_index - 1)).get_address(False, False))
                    self.__error_count += 1
                    self.__info_callback(msg)

        if -1 != cell_data.find('ERROR'):
            msg = ''
            # 零件号
            if (0 == (col_index % 3)):
                ecu_name = str(xw.Range((1, col_index)).value).strip()
                msg = '[{}]{} {} 零件号异常！请检查 {} 单元格！\n'.format(datetime.now(), 
                                                             vehicle_name, ecu_name, 
                                                             xw.Range((row[col_index -2].row, col_index)).get_address(False, False))

            # 软件号和软件版本号
            elif (1 == (col_index % 3)):
                ecu_name = str(xw.Range((1, col_index - 1)).value).strip()
                msg = '[{}]{} {} 软件号和软件版本号异常！请检查 {} 单元格！\n'.format(datetime.now(), 
                                                                   vehicle_name, ecu_name, 
                                                                   xw.Range((row[col_index -2].row, col_index)).get_address(False, False))

            # 配置码
            elif (2 == (col_index % 3)):
                ecu_name = str(xw.Range((1, col_index - 2)).value).strip()
                msg = '[{}]{} {} 配置码异常！请检查 {} 单元格！\n'.format(datetime.now(), 
                                                             vehicle_name, ecu_name, 
                                                             xw.Range((row[col_index -2].row, col_index)).get_address(False, False))

            self.__error_count += 1
            self.__info_callback(msg)

    def __summary(self) -> None:
        '''
        检测完成后生成总结信息字符串
        '''
        msg = ''
        if 0 == self.__error_count:
            msg = '[{}]检测完成，未发现异常！\n'.format(datetime.now())
        else:
            msg = '[{}]检测完成，总共发现 {} 处异常！\n'.format(datetime.now(), self.__error_count)

        self.__info_callback(msg)

    '''
配置文件翻译类定义文件
'''

import xlwings as xw
import pythoncom
from datetime import datetime

class Translator(object):
    '''
    配置文件翻译类，完成 Excel 配置文件生成新表格并翻译为英文的功能
    '''

    def __init__(self, config_file_path:str, trans_file_path:str, progress_callback=None, info_callback=None) -> None:
        '''
        构造函数

        参数说明：config_file_path，配置文件路径

                 trans_file_path，中英文对照表路径

                 progress_callback，外部传入的用于设置进度的函数，应当接收一个 float 型的参数，有效值为 0.0~100.0
                 函数原型：set_progress(progress:float=0.0)

                 info_callback，外部传入的用于显示信息的函数，应当接收一个 str 型的参数和一个 bool 型的参数
                 函数原型：show_info(info:str='', show_message_box:bool=false)
        '''
        super().__init__()
        pythoncom.CoInitialize()

        self.__config_file_path = config_file_path
        self.__trans_file_path = trans_file_path
        self.__progress_callbak = progress_callback
        self.__info_callback = info_callback
        self.__error_count = 0
        self.__trans_text = {}

    def __del__(self) -> None:
        '''
        析构函数，释放相关资源
        '''

        pythoncom.CoUninitialize()

    def start_translate(self) -> bool:
        '''
        开始进行文件翻译
        返回值：翻译未发生错误返回true；否则返回false
        '''
        if not self.__read_trans_text():
            return False

        if not self.__create_new_sheet():
            return False

        self.__summary()

        return True

    def __read_trans_text(self) -> bool:
        '''
        读取中英文翻译的文本
        '''
        if '' == self.__trans_file_path:
            return False

        app = xw.App(visible=False, add_book=False)
        book = app.books.open(self.__trans_file_path, read_only=True)
        sheet = book.sheets[0]
        sheet.activate()
        row_count = sheet.used_range.rows.count

        for row_index in range(2, row_count + 1):
            strCN = xw.Range((row_index, 1)).value
            strEN = xw.Range((row_index, 2)).value
            if not isinstance(strCN, str):
                continue

            self.__trans_text[strCN] = strEN


        book.close()
        app.quit()
        return True

    def __create_new_sheet(self) -> bool:
        '''
        复制原始配置表，并在新的表中将中文替换为英文
        '''
        if '' == self.__config_file_path:
            return False

        app = xw.App(visible=False, add_book=False)
        book = app.books.open(self.__config_file_path)
        sheet = book.sheets[0]
        sheet = sheet.copy(after=sheet) # 拷贝到第一个 sheet 之后
        sheet.activate()
        book.save()

        rows_count = sheet.used_range.rows.count
        cols_count = sheet.used_range.columns.count
        cells_to_trans = (cols_count - 2) / 3 + rows_count - 2
        cells_translated = 0

        # 更新第一行的控制器名称字符串
        for col_index in range(3, cols_count + 1):
            strCN = xw.Range((1, col_index)).value
            if not isinstance(strCN, str):
                cells_translated += 1
                continue

            if strCN in self.__trans_text:
                xw.Range((1, col_index)).value = self.__trans_text[strCN]
            else:
                msg = '[{}]{} 未找到对应的英文文本，请检查中英文对照表！\n'.format(datetime.now(), strCN)
                self.__info_callback(msg, False)
                self.__error_count += 1

            cells_translated += 1
            self.__progress_callbak(cells_translated / cells_to_trans * 100)

        # 更新第二列的车型名称字符串
        for row_index in range(3, rows_count + 1):
            strCN = xw.Range((row_index, 2)).value
            if not isinstance(strCN, str):
                cells_translated += 1
                continue

            if strCN in self.__trans_text:
                xw.Range((row_index, 2)).value = self.__trans_text[strCN]
            else:
                msg = '[{}]{} 未找到对应的英文文本，请检查中英文对照表！\n'.format(datetime.now(), strCN)
                self.__info_callback(msg, False)
                self.__error_count += 1

            cells_translated += 1
            self.__progress_callbak(cells_translated / cells_to_trans * 100)

        book.save()
        book.close()
        app.quit()
        return True

    def __summary(self) -> bool:
        '''
        翻译完成后生成总结信息字符串
        '''
        msg = ''
        if 0 == self.__error_count:
            msg = '[{}]翻译完成，未发现异常！\n'.format(datetime.now())
        else:
            msg = '[{}]翻译完成，总共发现 {} 处异常！\n'.format(datetime.now(), self.__error_count)

        self.__info_callback(msg)

        return True
