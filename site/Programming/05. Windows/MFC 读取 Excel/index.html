    参考链接：https://blog.csdn.net/cai_niaocainiao/article/details/81806928

    除Excel相关头文件外，必须包含自动化类头文件： 

    #include <afxdisp.h>        // MFC 自动化类

    在打开Excel之前应初始化COM组件： 

    HRESULT hr = CoInitialize(NULL);        //初始化COM组件，否则无法启动Excel程序
if (FAILED(hr))
    AfxMessageBox(_T("Failed to call Coinitialize()"));

    退出时应释放Excel相关资源： 

    // 释放对象

    m_ExcelBook.ReleaseDispatch();

    m_ExcelApp.Quit();                  // 退出 Excel 程序，注意这里要先退出，后释放

    CoUninitialize();       //必须与CoInitialize成对使用

    一个基础的Excel解析类： 

    #pragma once

/**************************************************************************************
*                               Excel文件解析类定义
*
* 类名：ExcelParser
*
* 用途：提供解析Excel文件需要的功能接口
*
* 用法：本类除构造函数和析构函数外，只有一个对外开放的接口 —— start_check，在外部只需实例化一个类
*      对象，然后调用 start_check 接口即可，所有检测的逻辑在本类内完成
**************************************************************************************/

#include "ExcelHeader/CApplication.h"
#include "ExcelHeader/CRange.h"
#include "ExcelHeader/CWorkbook.h"
#include "ExcelHeader/CWorkbooks.h"
#include "ExcelHeader/CWorksheet.h"
#include "ExcelHeader/CWorksheets.h"

class ExcelParser final
{
public:
    ExcelParser();
    ~ExcelParser();

    ExcelParser(const ExcelParser&) = delete;
    bool operator=(const ExcelParser&) = delete;

    bool open(const CString& strFilePath);                              //打开Excel文件，获取表格行数、列数
    void release();                                                     //释放资源，结束Excel程序

    long get_row_count() const { return m_lRowsCount; }                 //获取行数
    long get_col_count() const { return m_lColsCount; }                 //获取列数

    CString get_cell_str_data(long lRowIndex, long lColIndex);          //获取目标单元格中的字符串数据
    bool put_str_data_to_cell(long lRowIndex, long lColIndex, CString strData);         //将字符串数据写入目标单元格
    bool is_valid_cell(long lRowIndex, long lColIndex);                 //检测目标单元格是否有效

    static CString get_col_index_string(long lColIndex);                //将Excel列号转为字符形式
    void select_sheet_by_index(long lSheetIndex);                       //根据索引设置当前操作的Sheet
    void copy_sheet(long lSrcIndex);                                    //复制sheet内容

private:
    CApplication    m_ExcelApp;                 // Excel 应用程序接口
    CWorkbook       m_ExcelBook;                // 工作簿
    CWorkbooks      m_ExcelBooks;               // 工作簿集合
    CWorksheet      m_ExcelSheet;               // 工作表
    CWorksheets     m_ExcelSheets;              // 工作表集合
    CRange          m_ExcelRange;               // 用于对单元格进行操作
    CRange          m_UsedRange;

    long            m_lRowsCount;               //表格总行数
    long            m_lColsCount;               //表格总列数
    long            m_lSheetsCount;             //Sheet总数
};

/**************************************************************************************
* 函数名：ExcelParser::open
*
* 作用：打开Excel文件，获取表格行数、列数
*
* 参数：1.strFilePath ---- Excel文件路径
*
* 返回值：成功返回true；否则返回false
*
**************************************************************************************/
bool ExcelParser::open(const CString& strFilePath)
{
    if (!m_ExcelApp.CreateDispatch(_T("Excel.Application"), nullptr))
    {
        AfxMessageBox(_T("无法启动Excel服务器!"));
        return false;
    }

    LPDISPATCH lpDisp;
    COleVariant covOptional(static_cast<long>(DISP_E_PARAMNOTFOUND), VT_ERROR);
    m_ExcelBooks.AttachDispatch(m_ExcelApp.get_Workbooks());
    lpDisp = m_ExcelBooks.Open(strFilePath, covOptional, covOptional,
        covOptional, covOptional, covOptional, covOptional, covOptional,
        covOptional, covOptional, covOptional, covOptional, covOptional,
        covOptional, covOptional);

    //得到Workbook    
    m_ExcelBook.AttachDispatch(lpDisp);

    //得到Worksheets  
    m_ExcelSheets.AttachDispatch(m_ExcelBook.get_Worksheets());
    m_lSheetsCount = m_ExcelSheets.get_Count();

    //lpDisp = m_ExcelBook.get_ActiveSheet();
    //默认选择第一个sheet
    lpDisp = m_ExcelSheets.get_Item(COleVariant(long(1)));
    m_ExcelSheet.AttachDispatch(lpDisp);

    //获取行数、列数
    lpDisp = m_ExcelSheet.get_UsedRange();
    m_UsedRange.AttachDispatch(lpDisp);

    m_ExcelRange.AttachDispatch(m_UsedRange.get_Rows());
    m_lRowsCount = m_ExcelRange.get_Count();

    m_ExcelRange.AttachDispatch(m_UsedRange.get_Columns());
    m_lColsCount = m_ExcelRange.get_Count();

    return true;
}

/**************************************************************************************
* 函数名：ExcelParser::release
*
* 作用：释放资源，结束Excel程序
*
* 参数：无
*
* 返回值：无
*
**************************************************************************************/
void ExcelParser::release()
{
    // 释放对象
    m_ExcelBooks.ReleaseDispatch();
    m_ExcelBook.ReleaseDispatch();
    m_ExcelSheets.ReleaseDispatch();
    m_ExcelSheet.ReleaseDispatch();
    m_ExcelRange.ReleaseDispatch();
    m_UsedRange.ReleaseDispatch();

    m_ExcelApp.Quit();                  // 退出 Excel 程序，注意这里要先退出，后释放
    m_ExcelApp.ReleaseDispatch();
}

/**************************************************************************************
* 函数名：ExcelParser::get_cell_str_data
*
* 作用：获取目标单元格中的字符串数据，若单元格数据类型不是字符串，则返回空字符串
*
* 参数：1.lRowIndex ---- 目标单元格行号
*      2.lColIndex ---- 目标单元格列号
*
* 返回值：单元格中的字符串
*
**************************************************************************************/
CString ExcelParser::get_cell_str_data(long lRowIndex, long lColIndex)
{
    CString strData = _T("");
    if (lRowIndex < 1 || lRowIndex > m_lRowsCount ||
        lColIndex < 1 || lColIndex > m_lColsCount)
    {
        return strData;
    }

    COleVariant vResult;                //COleVariant类是对VARIANT结构的封装
    CRange cellRange;

    cellRange.AttachDispatch(m_ExcelSheet.get_Cells());
    cellRange.AttachDispatch(cellRange.get_Item(COleVariant(lRowIndex), COleVariant(lColIndex)).pdispVal);
    vResult = cellRange.get_Value2();
    if (VT_BSTR != vResult.vt)
    {
        cellRange.ReleaseDispatch();
        return strData;
    }

    strData = static_cast<CString>(vResult.bstrVal);
    cellRange.ReleaseDispatch();

    return strData;
}

/**************************************************************************************
* 函数名：ExcelParser::put_str_data_to_cell
*
* 作用：将字符串数据写入目标单元格
*
* 参数：1.lRowIndex ---- 目标单元格行号
*      2.lColIndex ---- 目标单元格列号
*      3.strData ---- 要写入的字符串
*
* 返回值：单元格为空返回false，否则返回true
*
**************************************************************************************/
bool ExcelParser::put_str_data_to_cell(long lRowIndex, long lColIndex, CString strData)
{
    if (lRowIndex < 1 || lRowIndex > m_lRowsCount ||
        lColIndex < 1 || lColIndex > m_lColsCount)
    {
        return false;
    }

    CRange cellRange;
    cellRange.AttachDispatch(m_UsedRange.get_Item(COleVariant(lRowIndex), COleVariant(lColIndex)).pdispVal);
    //COleVariant vResult = cellRange.get_Value2();
    cellRange.put_Value2(COleVariant(strData));
    cellRange.ReleaseDispatch();
    m_ExcelBook.Save();
    return true;
}

/**************************************************************************************
* 函数名：ExcelParser::is_valid_cell
*
* 作用：检测目标单元格是否有效（单元格为空视为无效）
*
* 参数：1.lRowIndex ---- 目标单元格行号
*      2.lColIndex ---- 目标单元格列号
*
* 返回值：单元格为空返回false，否则返回true
*
**************************************************************************************/
bool ExcelParser::is_valid_cell(long lRowIndex, long lColIndex)
{
    if (lRowIndex < 1 || lRowIndex > m_lRowsCount ||
        lColIndex < 1 || lColIndex > m_lColsCount)
    {
        return false;
    }

    COleVariant vResult;                //COleVariant类是对VARIANT结构的封装
    CRange cellRange;

    cellRange.AttachDispatch(m_ExcelSheet.get_Cells());
    cellRange.AttachDispatch(cellRange.get_Item(COleVariant(lRowIndex), COleVariant(lColIndex)).pdispVal);
    vResult = cellRange.get_Value2();
    cellRange.ReleaseDispatch();

    if (VT_EMPTY == vResult.vt)
    {
        return false;
    }

    return true;
}

/**************************************************************************************
* 函数名：ExcelParser::get_col_index_string
*
* 作用：将Excel列号转为字符形式
*
* 参数：1.lColIndex ---- 列号
*
* 返回值：列号的字符形式，如：1转为A，2转为B，27转为AA
* 
* 注意：此函数为类的静态成员函数
*
**************************************************************************************/
CString ExcelParser::get_col_index_string(long lColIndex)
{
    CString strCol = _T("");
    long index = lColIndex - 1;
    do
    {
        CString strTmp = _T("");
        if (!strCol.IsEmpty())
        {
            index--;
        }
        strTmp.Format(_T("%c"), index % 26 + 'A');
        strCol = strTmp + strCol;
        index /= 26;
    } while (index > 0);

    return strCol;
}

/**************************************************************************************
* 函数名：ExcelParser::select_sheet_by_index
*
* 作用：根据索引设置当前操作的Sheet
*
* 参数：1.lSheetIndex ---- 索引号，从1开始
*
* 返回值：无
* 
**************************************************************************************/
void ExcelParser::select_sheet_by_index(long lSheetIndex)
{

    //得到Worksheets  
    m_ExcelSheets.ReleaseDispatch();
    m_ExcelSheets.AttachDispatch(m_ExcelBook.get_Worksheets());
    m_lSheetsCount = m_ExcelSheets.get_Count();

    m_ExcelSheet.ReleaseDispatch();
    LPDISPATCH lpDisp;
    lpDisp = m_ExcelSheets.get_Item(COleVariant(lSheetIndex));
    m_ExcelSheet.AttachDispatch(lpDisp);

    //获取行数、列数
    m_UsedRange.ReleaseDispatch();
    lpDisp = m_ExcelSheet.get_UsedRange();
    m_UsedRange.AttachDispatch(lpDisp);

    m_ExcelRange.AttachDispatch(m_UsedRange.get_Rows());
    m_lRowsCount = m_ExcelRange.get_Count();

    m_ExcelRange.ReleaseDispatch();
    m_ExcelRange.AttachDispatch(m_UsedRange.get_Columns());
    m_lColsCount = m_ExcelRange.get_Count();
}

/**************************************************************************************
* 函数名：ExcelParser::copy_sheet
*
* 作用：根据索引指定的Sheet复制一份，新的 Worksheet 紧跟在原有的 Worksheet 之后
*
* 参数：1.lSrcIndex ---- 索引号，从1开始
*
* 返回值：无
*
**************************************************************************************/
void ExcelParser::copy_sheet(long lSrcIndex)
{
    LPDISPATCH lpDisp;
    CWorksheet sheetSrc;
    lpDisp = m_ExcelSheets.get_Item(COleVariant(lSrcIndex));
    sheetSrc.AttachDispatch(lpDisp);

    COleVariant temp;
    temp.pdispVal = sheetSrc.m_lpDispatch;
    temp.vt = VT_DISPATCH;
    //Worksheet 的 copy 函数的两个参数是移动的位置，在哪个 Worksheet 前，在哪个 Worksheet 后。
    //如果两个参数都不指定，则是复制到新的 Worksheet 中，这个新的 Worksheet 是所有 Worksheet 的最后一个
    //sheetSrc.Copy(vtMissing, temp);
    sheetSrc.Copy(vtMissing, _variant_t(sheetSrc));

    sheetSrc.ReleaseDispatch();
    m_ExcelBook.Save();
}
